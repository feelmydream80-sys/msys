from datetime import datetime, timedelta

import pandas as pd
import subprocess
import logging
import json

# 데이터베이스 테이블 이름
SERVICE_TABLE = 'tb_service_mst'
MONITOR_TABLE = 'tb_service_monitor'
HISTORY_TABLE = 'tb_service_monitor_hist'



class ServiceMonitor:
    def __init__(self, postgres_conn_id):
        self.postgres_conn_id = postgres_conn_id
        self.services = self._load_services()

    def _load_services(self):
        pg_hook = PostgresHook(postgres_conn_id=self.postgres_conn_id)
        connection = pg_hook.get_conn()
        cursor = connection.cursor()

        try:
            cursor.execute(f"""
                SELECT host, device_name, network, sw, port, service_desc, command 
                FROM {SERVICE_TABLE} 
                WHERE use_yn = 'Y'
            """)
            services = cursor.fetchall()
            df = pd.DataFrame(services, columns=['host', 'device_name', 'network', 'sw', 'port', 'service_desc', 'command'])
            return df.to_dict(orient='records')
        finally:
            cursor.close()
            connection.close()

    def check_service(self, command):
        try:
            command_func = command.split(' ')[0]
            if command_func == 'nc':
                result = subprocess.run(command, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
                return result.returncode == 0
            elif command_func == 'curl':
                result = subprocess.run(command, check=True, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
                result = json.loads(result.stdout.decode('utf-8'))
                totalResultCount = int(result['totalResultCount'])
                return totalResultCount == 1
        except Exception as e:
            logging.error(f"Error executing command '{command}': {e}")
            return False

    def check_all_services(self, **context):
        results = []
        for service in self.services:
            host = service['host']
            sw = service['sw']
            port = service['port']
            command = service['command']
            status = '정상' if self.check_service(command) else '비정상'
            results.append({
                'host': host,
                'sw': sw,
                'port': port,
                'status': status,
                'update_dt': datetime.now().strftime('%Y-%m-%d %H:%M:%S')
            })
        context['task_instance'].xcom_push(key='service_results', value=results)


    def update_monitor_table(self, **context):
        results = context['task_instance'].xcom_pull(key='service_results')
        if results is None:
            logging.error("No service results found")
            raise ValueError("No service results found")

        pg_hook = PostgresHook(postgres_conn_id=self.postgres_conn_id)
        connection = pg_hook.get_conn()
        cursor = connection.cursor()

        try:
            for result in results:
                host = result['host']
                sw = result['sw']
                port = result['port']
                status = result['status']
                update_dt = result['update_dt']

                cursor.execute(f"""
                    INSERT INTO {MONITOR_TABLE} (host, sw, port, update_dt, status)
                    VALUES (%s, %s, %s, %s, %s)
                    ON CONFLICT (host, sw, port) 
                    DO UPDATE SET status = EXCLUDED.status, update_dt = EXCLUDED.update_dt
                """, (host, sw, port, update_dt, status))

            connection.commit()
        except Exception as e:
            connection.rollback()
            logging.error(f"Error updating monitor table: {e}")
            raise
        finally:
            cursor.close()
            connection.close()

    def insert_history_table(self, **context):
        results = context['task_instance'].xcom_pull(key='service_results')
        if results is None:
            logging.error("No service results found")
            raise ValueError("No service results found")

        pg_hook = PostgresHook(postgres_conn_id=self.postgres_conn_id)
        connection = pg_hook.get_conn()
        cursor = connection.cursor()

        try:
            for result in results:
                host = result['host']
                sw = result['sw']
                port = result['port']
                status = result['status']
                create_dt = result['update_dt']

                cursor.execute(f"""
                    INSERT INTO {HISTORY_TABLE} (host, sw, port, create_dt, status)
                    VALUES (%s, %s, %s, %s, %s)
                """, (host, sw, port, create_dt, status))

            connection.commit()
        except Exception as e:
            connection.rollback()
            logging.error(f"Error inserting history table: {e}")
            raise
        finally:
            cursor.close()
            connection.close()

    def write_email(self, **context):
        results = context['task_instance'].xcom_pull(key='service_results')
        if results is None:
            logging.error("No service results found")
            raise ValueError("No service results found")

        pg_hook = PostgresHook(postgres_conn_id=self.postgres_conn_id)
        connection = pg_hook.get_conn()
        cursor = connection.cursor()
        mailing_cnt = 0
        mail_list = []
        
        try:
            for result in results:
                if result['status'] == '비정상':
                    print(result)
                    print('***************************')
                    host = result['host']
                    sw = result['sw']
                    port = result['port']
                    cursor.execute(f'''SELECT host, device_name, network, sw, port, service_desc, manager_email, email_dt
                                        FROM {SERVICE_TABLE} 
                                        WHERE manager_email is not null 
                                           AND host = %s 
                                           AND sw = %s 
                                           AND port = %s  
                                           AND use_yn='Y' LIMIT 1''',
                                         (host, sw, port))
                    obj = cursor.fetchone()
                    print(obj)
                    device_name = obj[1]
                    network = obj[2]
                    service_desc = obj[5]
                    manager_email = obj[6]
                    email_dt = obj[7]
                    now_dt = datetime.now()
                    if email_dt is None:
                        email_dt = now_dt
                    time_diff = now_dt - email_dt
                    # 메일이 최근 하루이내 발송 되지 않은 경우에만 발송
                    if time_diff.total_seconds() == 0 or time_diff.days >= 1:
                        logging.info('작성')  
                        # SER   
                        if manager_email is not None:
                            email_subject = f"[빅데이터 플랫폼] Service Alert: {host} - {device_name} - {sw} - {port} - {service_desc} is DOWN"
                            email_body = f"""
                            Service '{service_desc}' is down.<br/> <br/>
                            Host: {host} <br/> <br/>
                            Device Name: {device_name}<br/> <br/>
                            Network: {network}<br/> <br/>
                            Software: {sw}<br/> <br/>
                            Port: {port}<br/> <br/>
                            Status: {result['status']}<br/> <br/>
                            Time: {result['update_dt']}<br/> <br/>
                            """
                            temp_dic = {}
                            temp_dic['manager_email'] = manager_email
                            temp_dic['email_subject'] = email_subject
                            temp_dic['email_body'] = email_body
                            mail_list.append(temp_dic)

                            context['task_instance'].xcom_push(key='email_subject', value=email_subject)
                            context['task_instance'].xcom_push(key='email_body', value=email_body)
                            context['task_instance'].xcom_push(key='manager_email', value=manager_email)

                            mailing_cnt = mailing_cnt + 1
                        # SERVICE_TABLE 의 eamil_dt 현재 시간으로 update
                        cursor.execute(f"""
                            UPDATE {SERVICE_TABLE} a
                            SET EMAIL_DT = %s
                            WHERE a.HOST = %s AND a.SW = %s AND a.PORT = %s
                        """, (now_dt.strftime('%Y-%m-%d %H:%M:%S'), host, sw, port))
                        connection.commit()
                        
                    logging.info(f'{host}-{sw}-{service_desc}')
            
            context['task_instance'].xcom_push(key='mail_list', value=mail_list)

        except Exception as e:
            connection.rollback()
            logging.error(f"Error updating monitor table: {e}")
            raise


        finally:
            cursor.close()
            connection.close()

        if mailing_cnt == 0 :
            context['task_instance'].xcom_push(key='isEnded', value=True)
            return 'end_task'

    def send_emails(self, **context):
        results = context['task_instance'].xcom_pull(key='mail_list')
        print(results)
        
        if results is None :
            logging.error("No mail target")
            raise ValueError("No mail target") 

        for i, e in enumerate(results):
            task_id = f'send_mail_{i}'
            email_subject = e['email_subject']
            email_body = e['email_body']
            manager_email = e['manager_email']
            
            send_email_task = EmailOperator(
                task_id = task_id,
                to = manager_email,
                subject = email_subject,
                html_content = email_body,
                dag=context['dag']
            )
            send_email_task.execute(context)
            



default_args = {
    'owner': 'kepco_monitoring',
    'depends_on_past': False,
    'start_date': datetime(2024, 6, 10, 17, 00),
    'email_on_failure': False,
    'email_on_retry': False,
    'retries': 0,
}



with DAG('4_SERVICE_MONITORING',
         default_args=default_args,
         schedule_interval='*/10 * * * *',
         catchup=False) as dag:

    start = DummyOperator(
        task_id='start'
    )

    service_monitor = ServiceMonitor(postgres_conn_id='conn_postgres_bigdata_do')

    check_services = PythonOperator(
        task_id='check_services',
        python_callable=service_monitor.check_all_services,
        provide_context=True
    )

    update_monitor = PythonOperator(
        task_id='update_monitor',
        python_callable=service_monitor.update_monitor_table,
        provide_context=True
    )

    insert_history = PythonOperator(
        task_id='insert_history',
        python_callable=service_monitor.insert_history_table,
        provide_context=True
    )


    def branch_func(**context):
        results = context['task_instance'].xcom_pull(key='service_results')
        if results is None:
            logging.error("No service results found")
            raise ValueError("No service results found")
        for result in results:
            if result['status'] == '비정상':
                return 'write_email'
        return 'end_task'

    branch_task = BranchPythonOperator(
        task_id='branch_task',
        python_callable=branch_func,
        provide_context=True
    )

    write_email = PythonOperator(
        task_id='write_email',
        python_callable=service_monitor.write_email,
        provide_context=True
    )
    
    def branch_func_2(**context):
        flag = context['task_instance'].xcom_pull(key='isEnded')
        if flag:
            return 'end_task'
        else :        
            return 'write_email'

    branch_task_2 = BranchPythonOperator(
        task_id='branch_task_2',
        python_callable=branch_func_2,
        provide_context=True
    )

    with TaskGroup('send_emails') as send_group:
        send_email_task = PythonOperator(
            task_id = 'send_emails',
            python_callable=service_monitor.send_emails,
            provide_context=True,
        )
    #send_email = EmailOperator(
    #    task_id='send_email',
    #    to="{{ task_instance.xcom_pull(task_ids='write_email', key='manager_email') }}",
    #    subject="{{ task_instance.xcom_pull(task_ids='write_email', key='email_subject') }}",
    #    html_content="{{ task_instance.xcom_pull(task_ids='write_email', key='email_body') }}",
    #)



    end_task = DummyOperator(
        task_id='end_task'
    )

    start >> check_services >> update_monitor >> insert_history >> branch_task
    branch_task >> write_email >> branch_task_2 
    branch_task_2 >> send_group >> end_task
    branch_task_2 >> end_task
    branch_task >> end_task
